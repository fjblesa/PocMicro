--Move out reason

CREATE TABLE MOVEOUTREASON(
id GUID PRIMARY KEY,
name TEXT NOT NULL UNIQUE,
code TEXT NOT NULL UNIQUE,
active BOOLEAN,
distributed BOOLEAN
);

--Water Utility
CREATE TABLE  WATERUTILITY (
id GUID PRIMARY KEY,
code TEXT UNIQUE,
name TEXT UNIQUE,
identifier  TEXT NOT NULL,
active Boolean, 
);

--Commercial Site

CREATE TABLE  COMMERCIALSITE (
id GUID PRIMARY KEY,
code TEXT UNIQUE,
name TEXT UNIQUE,
active BOOLEAN,
waterutilityid GUID ,
FOREIGN KEY(waterutilityid)
REFERENCES WATERUTILITY(id)
);
